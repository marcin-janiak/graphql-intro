Error =/= Result

  • Internal Server Error
  • Unavailable In Country
  • Pet Not Found

union PetResult = Pet | UnavailableInCountry | PetNotFound
  

    query {
      __typename
      pet(species: "Cat") {
        ... on Pet {
          id
          name
        }
        ... on UnavailableInCountry {
          reason
          species
        }
      }
    }

Interfejsy


          interface Pet {
            id: ID!
            name: String
            legsCount: Int
          }
                            

          type Snake implements Pet {
            id: ID!
            name: String
            legsCount: Int
            length: Int!
          }
          
          type Cat implements Pet {
            id: ID!
            name: String
            legsCount: Int
            tailLength: Int!
          }

Union types


                            union SearchResult = Pet | Project | Country

Inline fragments


          query {
            search(filter: {name: {eq: "Luna"}}) {

                    id

                    ... on Cat {

                        length
                    }
                }
            
                ... on Project {

          }

Fragments


          {
          left: countries(filter: { code: { eq: "PL" } }) {
            ...comparisonFields
          }
          right: countries(filter: { code: { eq: "GB" } }) {
            ...comparisonFields
          }
          }
          
          fragment comparisonFields on Country {
          code
          name
          native
          }
